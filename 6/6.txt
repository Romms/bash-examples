Написати скрипт, який в якості першого параметра приймає команду, яку потрібно виконати, а в якості інших параметрів адреси серверів, на яких ця команда повинна виконатися.

#!/bin/bash
 
user=root
server=$*
nserver=$#
server=(`echo $server`)
 
while [ "$nserver" -gt 1 ] ; do
    ((nserver--))
    echo ${server[$nserver]}
    ssh ${user}@${server[$nserver]} $1
done
